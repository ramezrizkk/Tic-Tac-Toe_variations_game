42bool NumericalTTT_Board::is_win(Player<int>* player) {
43 // Winning condition: any row, column or diagonal sums to 15
44 for (int i = 0; i < 3; ++i) {
45 int rowSum = board[i][0] + board[i][1] + board[i][2];
46 int colSum = board[0][i] + board[1][i] + board[2][i];
47 if ((rowSum == 15 && board[i][0] && board[i][1] && board[i][2]) ||
48 (colSum == 15 && board[0][i] && board[1][i] && board[2][i]))
49 return true;
50 }
51
52 int diag1 = board[0][0] + board[1][1] + board[2][2];
53 int diag2 = board[0][2] + board[1][1] + board[2][0];
54
55 if ((diag1 == 15 && board[0][0] && board[1][1] && board[2][2]) ||
56 (diag2 == 15 && board[0][2] && board[1][1] && board[2][0]))
57 return true;
58
59 return false;
60}
