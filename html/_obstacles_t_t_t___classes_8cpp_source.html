43bool ObstaclesTTT_Board::check_line(int x, int y, int dx, int dy, char symbol) {
44 int count = 1; // Count the piece at (x, y) itself
45
46 // Check in positive direction (forward)
47 int nx = x + dx;
48 int ny = y + dy;
49 while (nx >= 0 && nx < rows && ny >= 0 && ny < columns && board[nx][ny] == symbol) {
50 count++;
51 nx += dx;
52 ny += dy;
53 }
54
55 // Check in negative direction (backward)
56 nx = x - dx;
57 ny = y - dy;
58 while (nx >= 0 && nx < rows && ny >= 0 && ny < columns && board[nx][ny] == symbol) {
59 count++;
60 nx -= dx;
61 ny -= dy;
62 }
63
64 return count >= 4;
65}

67bool ObstaclesTTT_Board::update_board(Move<char>* move) {

102bool ObstaclesTTT_Board::is_win(Player<char>* player) {
103 char sym = player->get_symbol();
104
105 if (check_line(last_x, last_y, 0, 1, sym)) return true; // Horizontal
106 if (check_line(last_x, last_y, 1, 0, sym)) return true; // Vertical
107 if (check_line(last_x, last_y, 1, 1, sym)) return true; // Diagonal
108 if (check_line(last_x, last_y, 1, -1, sym)) return true; // Anti-Diagonal
109
110 return false;
111}
