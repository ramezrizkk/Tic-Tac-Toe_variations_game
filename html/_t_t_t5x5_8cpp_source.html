24bool TTT5x5_Board::update_board(Move<char>* move)
25{
26 int x = move->get_x() ;
27 int y = move->get_y() ;
28 char mark = move->get_symbol();
29
30 // Validate move and apply if valid
31 if (!(x < 0 || x >= rows || y < 0 || y >= columns) &&
32 (board[x][y] == blank_symbol || mark == 0)) {
33
34 if (mark == 0) { // Undo move
35 n_moves--;
36 board[x][y] = blank_symbol;
37 }
38 else { // Apply move
39 n_moves++;
40 board[x][y] = toupper(mark);
41 }
42 return true;
43 }

47int TTT5x5_Board::count_triples(char sym) const
48{

51 auto check_triple = [&](int r, int c, int dr, int dc) {
52 return (board[r][c] == sym &&
53 board[r + dr][c + dc] == sym &&
54 board[r + 2 * dr][c + 2 * dc] == sym);
55 };
56
57 for (int r = 0; r < rows; ++r) {
58 for (int c = 0; c <= columns - 3; ++c) {
59 if (check_triple(r, c, 0, 1)) {
60 count++;
61 }
62 }
63 }
64

66 for (int c = 0; c < columns; ++c) {
67 if (check_triple(r, c, 1, 0)) {
68 count++;
69 }
70 }
71 }

74 for (int c = 0; c <= columns - 3; ++c) {

76 count++;
77 }
78 }
