106 bool update_board(Move<char>* move) override {
107 int row = move->get_x();
108 int col = move->get_y();
109 char mark = move->get_symbol();
110
111 // 1. Boundary Safety Check
112 if (row < 0 || row >= 9 || col < 0 || col >= 9) return false;
113
114 // 2. Occupancy Check
115 if (board[row][col] != '.') return false;
116
117 // 3. Zone Validation
118 int block_row = row / 3;
119 int block_col = col / 3;
120
121 // Ensure the zone isn't already claimed or full
122 if (zone_winners[block_row][block_col] != '.' || isZoneFull(block_row, block_col)) {
123 return false;
124 }
125
126 // 4. Place Mark
127 board[row][col] = mark;
128 n_moves++;
129
130 // 5. Update Zone Status
131 if (zone_winners[block_row][block_col] == '.') {
132 char result = scanForZoneWin(block_row, block_col);
133 if (result != '.') {
134 zone_winners[block_row][block_col] = result;
135 }
136 }
137
138 return true;

162 char scanForZoneWin(int block_row, int block_col) {
163 int base_r = block_row * 3;
164 int base_c = block_col * 3;
165
166 // Scan rows and columns within the zone
167 for (int k = 0; k < 3; k++) {
168 // Row check
169 if (board[base_r + k][base_c] != '.' &&
170 board[base_r + k][base_c] == board[base_r + k][base_c + 1] &&
171 board[base_r + k][base_c + 1] == board[base_r + k][base_c + 2])
172 return board[base_r + k][base_c];
173
174 // Column check
175 if (board[base_r][base_c + k] != '.' &&
176 board[base_r][base_c + k] == board[base_r + 1][base_c + k] &&
177 board[base_r + 1][base_c + k] == board[base_r + 2][base_c + k])
178 return board[base_r][base_c + k];
179 }
180
181 // Diagonal checks
182 char center = board[base_r + 1][base_c + 1];
183 if (center == '.') return '.'; // Optimization: if center is empty, diags can't win
184
185 if (board[base_r][base_c] == center && center == board[base_r + 2][base_c + 2])
186 return center;
187
188 if (board[base_r][base_c + 2] == center && center == board[base_r + 2][base_c])
189 return center;
190
191 return '.';
192 }
