21void ThreeWord_Board::load_dictionary() {
22 ifstream file("dic.txt");
23 if (!file.is_open()) {
24 cerr << "Error: Could not open dic.txt. Dictionary will be empty.\n";
25 return;
26 }
27
28 string word;
29 while (file >> word) {
30 // Convert word to uppercase
31 transform(word.begin(), word.end(), word.begin(), ::toupper);
32 if (word.length() == 3) {
33 dictionary.insert(word);
34 }
35 }
36 file.close();
37 cout << "Loaded " << dictionary.size() << " valid 3-letter words.\n";

83bool ThreeWord_Board::is_win(Player<char>* player) {

85 auto is_filled = [&](char a, char b, char c) {
86 return a != blank_symbol && b != blank_symbol && c != blank_symbol;
87 };
88
89 // Check rows
90 for (int i = 0; i < rows; ++i) {
91 if (is_filled(board[i][0], board[i][1], board[i][2])) {
92 string word(1, board[i][0]);
93 word += board[i][1];
94 word += board[i][2];
95 if (is_valid_word(word)) return true;
96 }
97 }
98
99 // Check columns
100 for (int i = 0; i < columns; ++i) {
101 if (is_filled(board[0][i], board[1][i], board[2][i])) {
102 string word(1, board[0][i]);
103 word += board[1][i];
104 word += board[2][i];
105 if (is_valid_word(word)) return true;
106 }
107 }
108
109 // Main diagonal
110 if (is_filled(board[0][0], board[1][1], board[2][2])) {
111 string word(1, board[0][0]);
112 word += board[1][1];
113 word += board[2][2];
114 if (is_valid_word(word)) return true;
115 }
116
117 // Anti-diagonal
118 if (is_filled(board[0][2], board[1][1], board[2][0])) {
119 string word(1, board[0][2]);
120 word += board[1][1];
121 word += board[2][0];
122 if (is_valid_word(word)) return true;
